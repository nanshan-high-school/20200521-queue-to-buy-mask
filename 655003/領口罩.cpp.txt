#include <iostream>
#include <queue>
using namespace std;

int main() {
	queue<string> name;

	int mask;
	
	cout << "口罩數量:";
	cin >> mask;

	cout << "請輸入名字(輸入n 結束):" << endl;

	do {
		string personname;
		cin >> personname;

		name.push(personname);
	} while (name.back() != "n");

	cout << "已領取:" << endl;

	for (int i = 0; mask >= 9; i++) {
		mask -= 9;

		cout << name.front() << endl;
		name.pop();
	}

	cout << "未領取:" << endl;
	
	while (name.front() != "n") {
		cout << name.front() << endl;
		name.pop();
	}
}
